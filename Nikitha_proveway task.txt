  <script>
    const basePrice = 24;

    function selectOption(element, units, discount) {
      document.querySelectorAll('.option').forEach(opt => opt.classList.remove('active'));
      element.classList.add('active');
      element.querySelector('input[type="radio"]').checked = true;

      const total = basePrice * units * (1 - discount / 100);
      document.getElementById('total').textContent = total.toFixed(2);
    }

    // Default: select second option
    window.onload = function() {
      const options = document.querySelectorAll('.option');
      selectOption(options[1], 2, 20);
      options[1].querySelector('input[type="radio"]').checked = true;
    };
